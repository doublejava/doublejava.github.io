# 按下面步骤一步一步走 ↓
    npm install -g vuepress
    新建目录，如F://myvuepress
    cd myvuepress

# 创建目录结构 ↓

    创建目录 docs，创建文件README.md ，注意大小写。
    进入docs目录，创建文件夹.vuepress，创建文件config.js 注意前面有个.
1
2

在config.js中复制如下代码 ↓

    module.exports = {
  title: '码路',
  port: '99',   //端口，默认8080
  description: '这里记录了我走过的码路，做不了大神，那我做个安静的路人好了...',
  // 注入到当前页面的 HTML <head> 中的标签
  head: [
    ['link', {
      rel: 'icon',
      href: '/favicon.ico'
    }], // 增加一个自定义的 favicon(网页标签的图标)
  ],
  base: '/', // 这是部署到github相关的配置 下面会讲
  markdown: {
    lineNumbers: true // 代码块显示行号
  },
  themeConfig: {
    sidebarDepth: 2, // e'b将同时提取markdown中h2 和 h3 标题，显示在侧边栏上。
    lastUpdated: 'Last Updated', // 文档更新时间：每个文件git最后提交的时间
    nav: [
      {
        text: 'vuepress搭建',
        link: '/vuepress/'
       },
    ],
  }
};
